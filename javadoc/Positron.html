/**
 * Defines program expressions.
 */
Start ::=
Import ::= ( ( command | prototype | function | <EOL> ) )* 
/*
////////////////////////////////////////////////////////////////////////
-----------------------------------------------------------------------
                              COMMANDS
-----------------------------------------------------------------------
\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\
*/
command ::= ( ( print <EOL> ) | ( println <EOL> ) | loop | conditional | ( expression <EOL> ) ) 
//* PRINT *************************************************************/
print ::= <PRINT> ( ( <LBRACE> expression ( "," expression )* <RBRACE> ) | expression ( "," expression )* ) 
//* SAY ***************************************************************/
println ::= <SAY> ( ( <LBRACE> expression ( "," expression )* <RBRACE> ) | expression ( "," expression )* ) 
//* WHILE *************************************************************/
loop ::= <WHILE> <LBRACE> expression ( <LOGICAL> expression )* <RBRACE> ( ( block ) | <EOL> ) 
//* IF-ELSIF-ELSE *****************************************************/
conditional ::= <IF> <LBRACE> expression ( <LOGICAL> expression )* <RBRACE> block ( <ELSIF> <LBRACE> expression ( <LOGICAL> expression )* <RBRACE> block )* ( <ELSE> block )? 
/*
////////////////////////////////////////////////////////////////////////
-----------------------------------------------------------------------
                              SYSCALLS
-----------------------------------------------------------------------
\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\
*/
syscall ::= ( drag | drop | look | ask | fire | hit | move ) 
//* DRAG **************************************************************/
drag ::= <DRAG> ( ( <LBRACE> expression <RBRACE> ) | expression ) 
//* DROP **************************************************************/
drop ::= <DROP> ( ( <LBRACE> expression <RBRACE> ) | expression ) 
//* LOOK **************************************************************/
look ::= <LOOK> <LBRACE> expression <RBRACE> 
//* ASK ***************************************************************/
ask ::= <ASK> ( ( <LBRACE> expression <RBRACE> ) | expression ) 
//* ATTACKS ***********************************************************/
hit ::= <HIT> <LBRACE> expression <RBRACE>
fire ::= <FIRE> <LBRACE> expression <RBRACE> 
//* MOVE **************************************************************/
move ::= <MOVE> <LBRACE> expression <RBRACE> 
/*
////////////////////////////////////////////////////////////////////////
-----------------------------------------------------------------------
                             FUNCTIONS
-----------------------------------------------------------------------
\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\
*/
header ::= <TYPE> <WORD> <LBRACE> ( <TYPE> ( <WORD> )? ( "," <TYPE> ( <WORD> )? )* )? <RBRACE>
prototype ::= <DEF> header <EOL>
function ::= header block
block ::= <LCURLY> ( ( <RET> expression <EOL> ) | ( command ) )* <RCURLY> 
/*
////////////////////////////////////////////////////////////////////////
-----------------------------------------------------------------------
                          AUXILIAR METHODS
-----------------------------------------------------------------------
\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\
*/
auxMethods ::= ( toCoord | toNumI | toNumJ )
toCoord ::= <TOCOORD> <LBRACE> expression <RBRACE>
toNumI ::= <TONUM_I> <LBRACE> expression <RBRACE>
toNumJ ::= <TONUM_J> <LBRACE> expression <RBRACE> 
/*
////////////////////////////////////////////////////////////////////////
-----------------------------------------------------------------------
                            EXPRESSIONS
-----------------------------------------------------------------------
\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\\
*/

/**
 * Expression is a comparison of Arit's.
 */
expression ::= arit ( <CMP> arit )* 
/**
 * Arit is a sum of terms.
 */
arit ::= term ( <SIGN> term )* 
/**
 * Term is a product of elements.
 */
term ::= element ( <OP> element )* 
/**
 * Element is either a factor or a signed factor.
 */
element ::= ( factor )
| ( <SIGN> factor ) 
/**
 * Factor is either a attribution, a variable, 
 * a number or a braced expression.
 */
factor ::= ( ( ( <SCOPE> )? <WORD> ( <ATTR> expression ) ) | ( <MAGIC> ( <ATTR> expression ) ) )
| ( <WORD> <LBRACE> ( expression ( "," expression )* )? <RBRACE> )
| <MAGIC>
| <WORD>
| ( coord )
| ( syscall )
| ( auxMethods )
| ( <ITEM> )
| ( <WAY> )
| ( <STRING> )
| ( <NUMBER> )
| <NULL>
| ( <LBRACE> expression <RBRACE> )
coord ::= "[" expression "," expression "]"
